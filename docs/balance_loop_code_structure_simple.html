平衡环代码结构拆解：PD 内环先独立

这是一套简化版代码架构。核心原则是：ROS 通信、状态估计、安全限幅、轮子输出和 PD 控制器分开。 后续要加入速度环、位置环、yaw 环或串级 PID 时，只扩展控制层，不推翻节点主流程。

设计目标

第一版只解决“站住”的问题：用 IMU 的 pitch 和 pitch_rate 计算左右轮同向角速度。它不处理速度目标、不处理位置目标、不处理 yaw 转向。

PD 控制独立

BalancePDController 只接收状态和参数，只输出一个前后方向轮速。

可替换

ROS 节点只编排

节点负责订阅、定时器、发布和参数加载，不把控制算法写死在回调里。

清晰边界

预留串级控制

速度环以后只需要输出 pitch_target，再交给姿态 PD 内环。

可扩展

分层结构

ROS I/O 层
订阅 /imu/data，发布 /wheel_velocity_controller/commands 和 debug。
状态估计层
将 IMU quaternion 转成 roll/pitch/yaw，整理成 BalanceState。
控制算法层
当前只有 BalancePDController，后续加入速度环和位置环。
安全层
处理 IMU 超时、跌倒角、速度限幅、加速度限幅。
输出层
将一个平衡轮速 balance_w 转换成左右轮数组 [left_w, right_w]。

PD 内环

BalancePDController 不知道 ROS，不知道轮子话题，也不处理限幅。 它只做一件事：根据 pitch 和 pitch_rate 输出一个平衡轮速。

balance_pd.py

#!/usr/bin/env python3

from dataclasses import dataclass

from balance_types import BalanceState


@dataclass
class BalancePDConfig:
    kp_pitch: float = 2.0
    kd_pitch: float = 0.12
    pitch_target: float = 0.0
    output_sign: float = 1.0


class BalancePDController:
    def __init__(self, config: BalancePDConfig):
        self.config = config

    def compute(self, state: BalanceState) -> float:
        pitch_error = state.pitch - self.config.pitch_target

        # D 项直接使用 IMU gyro_y，不对 pitch 做差分，避免放大噪声。
        raw_output = (
            self.config.kp_pitch * pitch_error
            - self.config.kd_pitch * state.pitch_rate
        )

        return self.config.output_sign * raw_output

注意：PD 输出的正负方向不一定第一次就对。如果车向前倒时轮子反向跑，只改 output_sign，不要同时改 kp_pitch 或 kd_pitch。

节点主流程

节点只做编排：读取最新状态、调用控制器、经过安全限幅、混合左右轮、发布命令。

balance_controller_node.py 主循环骨架

def on_timer(self):
    state = self.state_estimator.latest_state()

    if not self.safety.is_state_valid(state):
        self.publish_wheels(0.0, 0.0)
        return

    # 第一阶段：pitch PD 内环直接输出平衡轮速。
    balance_w = self.balance_pd.compute(state)

    # 预留：未来 turn_w 来自 yaw 控制或 cmd_vel。
    turn_w = 0.0

    left_w, right_w = self.wheel_mixer.mix(balance_w, turn_w)
    left_w, right_w = self.safety.limit_command(left_w, right_w)

    if self.enabled:
        self.publish_wheels(left_w, right_w)
    else:
        self.publish_wheels(0.0, 0.0)

    self.publish_debug(state, balance_w, left_w, right_w)

后续串级扩展

串级 PID 的关键是：外环不要直接控制轮子。外环应该输出内环目标。 对平衡车来说，速度环输出 pitch_target，姿态环再输出轮速。

第一阶段：只有姿态内环

pitch_target = 0.0
balance_w = balance_pd.compute(state)

第二阶段：速度环 + 姿态环

velocity_error = target_velocity - measured_velocity
pitch_target = velocity_pid.compute(velocity_error, dt)
pitch_target = clamp(pitch_target, -max_pitch_target, max_pitch_target)

state.pitch_target = pitch_target
balance_w = balance_pd.compute(state)

第三阶段：位置环 + 速度环 + 姿态环

position_error = target_position - measured_position
target_velocity = position_pid.compute(position_error, dt)
target_velocity = clamp(target_velocity, -max_target_velocity, max_target_velocity)

velocity_error = target_velocity - measured_velocity
pitch_target = velocity_pid.compute(velocity_error, dt)
pitch_target = clamp(pitch_target, -max_pitch_target, max_pitch_target)

state.pitch_target = pitch_target
balance_w = balance_pd.compute(state)

yaw 转向混合

turn_w = yaw_controller.compute(yaw_error, yaw_rate)

left_w = balance_w - turn_w
right_w = balance_w + turn_w

推荐扩展顺序：先姿态 PD 站住，再加轮速阻尼，再加速度环，最后加位置环和 yaw 转向。

src/balance_car_control/src/attitude_estimator.py

#!/usr/bin/env python3

import math

from balance_types import BalanceState


def quaternion_to_euler(x, y, z, w):
    sinr_cosp = 2.0 * (w * x + y * z)
    cosr_cosp = 1.0 - 2.0 * (x * x + y * y)
    roll = math.atan2(sinr_cosp, cosr_cosp)

    sinp = 2.0 * (w * y - z * x)
    if abs(sinp) >= 1.0:
        pitch = math.copysign(math.pi / 2.0, sinp)
    else:
        pitch = math.asin(sinp)

    siny_cosp = 2.0 * (w * z + x * y)
    cosy_cosp = 1.0 - 2.0 * (y * y + z * z)
    yaw = math.atan2(siny_cosp, cosy_cosp)

    return roll, pitch, yaw


class AttitudeEstimator:
    def __init__(self):
        self._latest_state = None

    def update_from_imu(self, imu_msg, stamp_sec):
        q = imu_msg.orientation
        _, pitch, yaw = quaternion_to_euler(q.x, q.y, q.z, q.w)

        self._latest_state = BalanceState(
            stamp_sec=stamp_sec,
            pitch=pitch,
            pitch_rate=imu_msg.angular_velocity.y,
            yaw=yaw,
        )

    def latest_state(self):
        return self._latest_state

src/balance_car_control/src/safety_limiter.py

#!/usr/bin/env python3

from dataclasses import dataclass

from balance_types import WheelCommand


def clamp(value, lower, upper):
    return max(lower, min(upper, value))


@dataclass
class SafetyConfig:
    max_wheel_velocity: float = 3.0
    max_wheel_accel: float = 60.0
    fall_angle_rad: float = 0.61
    imu_timeout_sec: float = 0.05


class SafetyLimiter:
    def __init__(self, config: SafetyConfig):
        self.config = config
        self._last_left = 0.0
        self._last_right = 0.0

    def reset(self):
        self._last_left = 0.0
        self._last_right = 0.0

    def is_state_valid(self, state, now_sec):
        if state is None:
            return False

        if now_sec - state.stamp_sec > self.config.imu_timeout_sec:
            return False

        if abs(state.pitch) > self.config.fall_angle_rad:
            return False

        return True

    def _limit_one(self, target, previous, dt):
        velocity_limit = abs(self.config.max_wheel_velocity)
        saturated = clamp(target, -velocity_limit, velocity_limit)

        max_delta = abs(self.config.max_wheel_accel) * max(dt, 0.0)
        delta = clamp(saturated - previous, -max_delta, max_delta)
        return previous + delta

    def limit_command(self, command: WheelCommand, dt):
        left = self._limit_one(command.left, self._last_left, dt)
        right = self._limit_one(command.right, self._last_right, dt)

        self._last_left = left
        self._last_right = right

        return WheelCommand(left=left, right=right)

src/balance_car_control/src/balance_controller_node.py

#!/usr/bin/env python3

import math

import rclpy
from rclpy.node import Node
from sensor_msgs.msg import Imu
from std_msgs.msg import Float64MultiArray

from attitude_estimator import AttitudeEstimator
from balance_pd import BalancePDConfig, BalancePDController
from safety_limiter import SafetyConfig, SafetyLimiter
from wheel_mixer import WheelMixer


class BalanceControllerNode(Node):
    def __init__(self):
        super().__init__("balance_controller_node")

        self.declare_parameter("enabled", False)
        self.declare_parameter("imu_topic", "/imu/data")
        self.declare_parameter("command_topic", "/wheel_velocity_controller/commands")
        self.declare_parameter("debug_topic", "/balance_controller/debug")
        self.declare_parameter("control_rate", 200.0)

        self.declare_parameter("pitch_target", 0.0)
        self.declare_parameter("kp_pitch", 2.0)
        self.declare_parameter("kd_pitch", 0.12)
        self.declare_parameter("output_sign", 1.0)

        self.declare_parameter("max_wheel_velocity", 3.0)
        self.declare_parameter("max_wheel_accel", 60.0)
        self.declare_parameter("fall_angle_rad", 0.61)
        self.declare_parameter("imu_timeout_sec", 0.05)

        self.enabled = bool(self.get_parameter("enabled").value)

        pd_config = BalancePDConfig(
            kp_pitch=float(self.get_parameter("kp_pitch").value),
            kd_pitch=float(self.get_parameter("kd_pitch").value),
            pitch_target=float(self.get_parameter("pitch_target").value),
            output_sign=float(self.get_parameter("output_sign").value),
        )
        safety_config = SafetyConfig(
            max_wheel_velocity=float(self.get_parameter("max_wheel_velocity").value),
            max_wheel_accel=float(self.get_parameter("max_wheel_accel").value),
            fall_angle_rad=float(self.get_parameter("fall_angle_rad").value),
            imu_timeout_sec=float(self.get_parameter("imu_timeout_sec").value),
        )

        self.estimator = AttitudeEstimator()
        self.balance_pd = BalancePDController(pd_config)
        self.safety = SafetyLimiter(safety_config)
        self.wheel_mixer = WheelMixer()

        self.command_pub = self.create_publisher(
            Float64MultiArray,
            self.get_parameter("command_topic").value,
            10,
        )
        self.debug_pub = self.create_publisher(
            Float64MultiArray,
            self.get_parameter("debug_topic").value,
            10,
        )
        self.create_subscription(
            Imu,
            self.get_parameter("imu_topic").value,
            self.on_imu,
            20,
        )

        self.last_control_time = self.get_clock().now()
        control_rate = float(self.get_parameter("control_rate").value)
        self.create_timer(1.0 / control_rate, self.on_timer)

        self.get_logger().warn(
            "Balance controller started. enabled={} kp={:.3f} kd={:.3f}".format(
                self.enabled,
                pd_config.kp_pitch,
                pd_config.kd_pitch,
            )
        )

    def now_sec(self):
        return self.get_clock().now().nanoseconds / 1e9

    def on_imu(self, msg):
        self.estimator.update_from_imu(msg, self.now_sec())

    def publish_wheels(self, left, right):
        self.command_pub.publish(Float64MultiArray(data=[left, right]))

    def stop_wheels(self):
        self.safety.reset()
        self.publish_wheels(0.0, 0.0)

    def publish_debug(self, state, balance_output, left, right):
        if state is None:
            return

        self.debug_pub.publish(Float64MultiArray(data=[
            state.pitch,
            state.pitch_rate,
            self.balance_pd.config.pitch_target,
            balance_output,
            left,
            right,
            1.0 if self.enabled else 0.0,
            state.yaw,
        ]))

    def on_timer(self):
        now = self.get_clock().now()
        dt = (now - self.last_control_time).nanoseconds / 1e9
        self.last_control_time = now

        state = self.estimator.latest_state()
        if not self.safety.is_state_valid(state, self.now_sec()):
            if state is not None and abs(state.pitch) > self.safety.config.fall_angle_rad:
                self.get_logger().warn(
                    "Fall angle reached: pitch={:.2f} deg".format(math.degrees(state.pitch)),
                    throttle_duration_sec=1.0,
                )
            self.stop_wheels()
            return

        balance_w = self.balance_pd.compute(state)

        # 第一阶段不做转向。未来 yaw 环或 cmd_vel 可以在这里生成 turn_w。
        turn_w = 0.0
        mixed_command = self.wheel_mixer.mix(balance_w, turn_w)
        safe_command = self.safety.limit_command(mixed_command, dt)

        if self.enabled:
            self.publish_wheels(safe_command.left, safe_command.right)
        else:
            self.stop_wheels()
            safe_command.left = 0.0
            safe_command.right = 0.0

        self.publish_debug(
            state=state,
            balance_output=balance_w,
            left=safe_command.left,
            right=safe_command.right,
        )


def main():
    rclpy.init()
    node = BalanceControllerNode()
    try:
        rclpy.spin(node)
    finally:
        node.stop_wheels()
        node.destroy_node()
        rclpy.shutdown()


if __name__ == "__main__":
    main()
